//header <script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js"></script> <script src="root/js/leaflet.js"></script> <link rel="stylesheet" type="text/css" href="root/css/leaflet.css"> //body <center><div id="map" style="width: 400px;height: 400px"></div></center> //end of the body <script> //map basic setting var map = L.map('map',{ center: [46.6863481,7.863205], zoom: 5 }); //add open street map layer to map div L.tileLayer('http://{s}.tile.osm.org/{z}/{x}/{y}.png', { attribution: '&copy; <a href="http://osm.org/copyright">OpenStreetMap</a> contributors', maxZoom: 18, id: 'mapbox.streets', }).addTo(map); //define map icon var pointIcon = L.icon({ iconUrl: './root/images/startmark.png', iconSize: [35, 35], // size of the icon popupAnchor: [0, 0] }); //add icon to map L.marker([46.685013 , 7.863804], {icon: pointIcon}).addTo(map); </script> for creating a popup you need to attach these codes to icon :
//add icon with popup L.marker([46.685013 , 7.863804], {icon: pointIcon}).addTo(map).bindPopup("<span>This Is a POPUP</span>");

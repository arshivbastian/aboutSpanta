Location circle on map

First Look :

we can use circle instead of icon on map

First you need mother code :

//header <script src="https://ajax.googleapis.com/ajax/libs/jquery/3.3.1/jquery.min.js"></script> <script src="root/js/leaflet.js"></script> <link rel="stylesheet" type="text/css" href="root/css/leaflet.css"> //body <center><div id="map" style="width: 400px;height: 400px"></div></center> //end of the body <script> //map basic setting var map = L.map('map',{ center: [0,0], zoom: 5 }); //add open street map layer to map div L.tileLayer('http://{s}.tile.osm.org/{z}/{x}/{y}.png', { attribution: '&copy; <a href="http://osm.org/copyright">OpenStreetMap</a> contributors', maxZoom: 18, id: 'mapbox.streets', }).addTo(map); </script> the extera code in end of body script //blue circle $("#blue").click (function() { var circle = L.circle([46.685182 , 7.865076], { color: 'blue', fillColor: '#6536c2', fillOpacity: 0.5, radius: 10 }).addTo(map) }); //red circle $("#red").click (function() { var circle = L.circle([46.685013 , 7.863804], { color: 'red', fillColor: '#990000', fillOpacity: 0.5, radius: 10 }).addTo(map) });
